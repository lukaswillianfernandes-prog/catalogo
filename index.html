(async () => { const BASE = 'https://royal-sports.x.yupoo.com'; const delay = ms => new Promise(r => setTimeout(r, ms)); const IGNORAR = ['Promoções','Kids Tracksuit','Website shopping tutorial', 'Casual T-shirt','Windbreaker','Patch !','sock','Shorts','SIZE', 'Call Me Whatsapp','Call Me →','Player','Total classification', 'NBA Silk version','F1','NFL','MLB','NHL','Hat',"kids'",'Woman', 'retro style 2','Baby jersey size:9-12','Kings League']; const results = []; let totalImgs = 0; console.log('Buscando categorias...'); const catRes = await fetch(BASE + '/categories', {credentials:'include'}); const catHtml = await catRes.text(); const cats = []; const catParts = catHtml.split('href="/categories/'); for (let i = 1; i < catParts.length; i++) { const idEnd = catParts[i].indexOf('"'); const id = catParts[i].substring(0, idEnd); const nameStart = catParts[i].indexOf('>') + 1; const nameEnd = catParts[i].indexOf('<'); const name = catParts[i].substring(nameStart, nameEnd).trim(); if (id && id !== '0' && name && !cats.find(c => c.id === id) && !IGNORAR.includes(name)) { cats.push({id, name}); } } console.log('Categorias: ' + cats.length); for (const cat of cats) { console.log('Categoria: ' + cat.name); // Busca subcategorias const catPageRes = await fetch(BASE + '/categories/' + cat.id + '?uid=1', {credentials:'include'}); const catPageHtml = await catPageRes.text(); // Extrai subcategorias const subcats = []; const subParts = catPageHtml.split('href="/categories/'); for (let i = 1; i < subParts.length; i++) { const idEnd = subParts[i].indexOf('"'); const subId = subParts[i].substring(0, idEnd); const nameStart = subParts[i].indexOf('>') + 1; const nameEnd = subParts[i].indexOf('<'); const subName = subParts[i].substring(nameStart, nameEnd).trim(); if (subId && subId !== cat.id && subId !== '0' && subName && !subcats.find(s => s.id === subId) && !cats.find(c => c.id === subId)) { subcats.push({id: subId, name: subName, parentCat: cat.name}); } } // Se tem subcategorias, usa elas; senão usa a categoria principal const targets = subcats.length > 0 ? subcats.map(s => ({id: s.id, catName: cat.name + ' > ' + s.name})) : [{id: cat.id, catName: cat.name}]; if(subcats.length > 0) console.log(' Subcategorias: ' + subcats.map(s=>s.name).join(', ')); for (const target of targets) { let page = 1, hasMore = true; const albums = []; while (hasMore) { const url = BASE + '/categories/' + target.id + '?page=' + page + '&uid=1'; const res = await fetch(url, {credentials:'include'}); const html = await res.text(); const parts = html.split('/albums/'); const ids = []; for (let i = 1; i < parts.length; i++) { let id = ''; for (let j = 0; j < parts[i].length; j++) { if (parts[i][j] >= '0' && parts[i][j] <= '9') id += parts[i][j]; else break; } if (id && !ids.includes(id)) ids.push(id); } if (ids.length === 0) { hasMore = false; break; } ids.forEach(id => { if (!albums.find(a => a.id === id)) albums.push({id}); }); hasMore = html.includes('page=' + (page + 1)) && page < 100; page++; await delay(400); } console.log(' ' + target.catName + ': ' + albums.length + ' albuns'); for (const album of albums) { try { const aUrl = BASE + '/albums/' + album.id + '?uid=1'; const aRes = await fetch(aUrl, {credentials:'include'}); const aHtml = await aRes.text(); // Nome via data-name let albumName = ''; const dn = aHtml.indexOf('data-name="'); if (dn > -1) { const dnStart = dn + 11; const dnEnd = aHtml.indexOf('"', dnStart); if (dnEnd > dnStart) albumName = aHtml.substring(dnStart, dnEnd).trim(); } // Fallback: title if (!albumName) { const t1 = aHtml.indexOf(''); const t2 = aHtml.indexOf(''); if (t1 > -1 && t2 > t1) albumName = aHtml.substring(t1+7, t2).split('|')[0].trim(); } if (!albumName) albumName = 'Album ' + album.id; albumName = albumName.replace(/<[^>]+>/g,'').trim(); // Imagens /large. apenas const imgUrls = []; const srcParts = aHtml.split('src="'); for (let i = 1; i < srcParts.length; i++) { const end = srcParts[i].indexOf('"'); const u = srcParts[i].substring(0, end); if (u.indexOf('photo.yupoo.com') > -1 && u.indexOf('/large.') > -1) { if (!imgUrls.includes(u)) imgUrls.push(u); } } // Usa cat sem subcategoria (ex: "Brasileiro Série A" sem "> Grêmio") const catFinal = cat.name; if (imgUrls.length > 0 && !results.find(r => r.albumId === album.id)) { results.push({cat: catFinal, album: albumName, albumId: album.id, images: imgUrls}); totalImgs += imgUrls.length; } await delay(200); } catch(e) { console.warn('Erro album ' + album.id + ': ' + e.message); } } } } console.log('Concluido! ' + results.length + ' albuns, ' + totalImgs + ' imagens.'); const blob = new Blob([JSON.stringify(results, null, 2)], {type:'application/json'}); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = 'yupoo_catalog_completo.json'; a.click(); window._yupooData = results; console.log('Arquivo yupoo_catalog_completo.json baixado!'); })();
